    <script>
        // แก้ไขลิงก์เชื่อมหลังบ้านและ LIFF ID ของคุณที่นี่
        const BACKEND_URL = "https://script.google.com/macros/s/AKfycby8tQc_HBPqAODr-tA7bO3fVlzZTylsWa8CMVmqJepdeuqoKwnKeQeQt_XDucjmabvS/exec"; 
        let userId = "test_line_id_123"; 

        window.onload = function() {
            initializeLiff();
        };

        function initializeLiff() {
            liff.init({ liffId: "2010178370-xHiJkPvz" }) 
                .then(() => {
                    if (liff.isLoggedIn()) {
                        liff.getProfile().then(profile => {
                            userId = profile.userId;
                            checkMemberStatus();
                        });
                    } else {
                        liff.login();
                    }
                })
                .catch((err) => {
                    console.log("เปิดในโหมดทดสอบ (Local Test)");
                    checkMemberStatus(); 
                });
        }

        function checkMemberStatus() {
            fetch(`${BACKEND_URL}?action=getProfile&lineId=${userId}`)
                .then(res => res.json())
                .then(data => {
                    document.getElementById("loading").classList.add("hidden");
                    if (data.exists) {
                        document.getElementById("userName").innerText = data.name;
                        document.getElementById("userPoints").innerText = data.points.toLocaleString();
                        document.getElementById("userTier").innerText = data.tier;
                        document.getElementById("userTier").className = `tier ${data.tier}`;
                        document.getElementById("userDiscount").innerText = data.discount; // แสดงผลส่วนลด %
                        document.getElementById("dashboard").classList.remove("hidden");
                    } else {
                        document.getElementById("registerForm").classList.remove("hidden");
                    }
                });
        }

        function registerUser() {
            const name = document.getElementById("regName").value;
            const phone = document.getElementById("regPhone").value;
            if(!name || !phone) return alert("กรุณากรอกข้อมูลให้ครบถ้วน");

            document.getElementById("registerForm").classList.add("hidden");
            document.getElementById("loading").classList.remove("hidden");

            fetch(BACKEND_URL, {
                method: "POST",
                body: JSON.stringify({ action: "register", lineId: userId, name: name, phone: phone })
            })
            .then(res => res.json())
            .then(data => {
                if(data.status === "success") {
                    checkMemberStatus();
                }
            });
        }
