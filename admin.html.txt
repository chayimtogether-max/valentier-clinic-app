    <div class="box">
        <h2>ระบบเติมพ้อยท์ลูกค้า</h2>
        <input type="tel" id="searchPhone" placeholder="กรอกเบอร์โทรศัพท์ลูกค้า">
        <input type="number" id="pointsToAdd" placeholder="จำนวนคะแนนที่ต้องการเพิ่ม">
        <button onclick="addPoints()">ยืนยันการเพิ่มคะแนน</button>
    </div>

    <script>
        const BACKEND_URL = "https://script.google.com/macros/s/AKfycby8tQc_HBPqAODr-tA7bO3fVlzZTylsWa8CMVmqJepdeuqoKwnKeQeQt_XDucjmabvS/exec";

        function addPoints() {
            const phone = document.getElementById("searchPhone").value;
            const points = document.getElementById("pointsToAdd").value;
            if(!phone || !points) return alert("กรุณากรอกข้อมูลให้ครบ");

            fetch(BACKEND_URL, {
                method: "POST",
                body: JSON.stringify({ action: "addPoints", phone: phone, points: points })
            })
            .then(res => res.json())
            .then(data => {
                if(data.status === "success") {
                    alert(`เติมคะแนนสำเร็จ! คะแนนใหม่ของลูกค้าคือ: ${data.newPoints} คะแนน`);
                    document.getElementById("pointsToAdd").value = "";
                } else {
                    alert("เกิดข้อผิดพลาด: " + data.message);
                }
            });
        }
    </script>
